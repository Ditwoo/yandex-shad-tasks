package ua.yandex.shad.tempseries;

public class TempSummaryStatistics {

    private final double EPSILON = 0.00000001;

    public double avgTemp;
    public double devTemp;
    public double minTemp;
    public double maxTemp;

    public TempSummaryStatistics() {
        avgTemp = 0.0;
        devTemp = 0.0;
        minTemp = 0.0;
        maxTemp = 0.0;
    }

    public TempSummaryStatistics(double avg, double dev, double min, double max) {
        avgTemp = avg;
        devTemp = dev;
        minTemp = min;
        maxTemp = max;
    }

    @Override
    public boolean equals(Object obj) {
        TempSummaryStatistics value = (TempSummaryStatistics) obj;
        return (Math.abs(this.avgTemp - value.avgTemp) < EPSILON
                && Math.abs(this.devTemp - value.devTemp) < EPSILON
                && Math.abs(this.minTemp - value.minTemp) < EPSILON
                && Math.abs(this.maxTemp - value.maxTemp) < EPSILON);
    }
}
